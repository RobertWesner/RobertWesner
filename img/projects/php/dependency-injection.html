Fun PSR-11 with autowiring


readonly class Foo
{
    public function __construct(
        // this will be autowired with all dependencies automagically
        private Bar $bar,
        #[AutowireValue(1234)]
        private int $scalar,
        #[AutowireEnv(__DIR__ . '/../../.env', 'TEST')]
        private string $envTest,
        #[AutowireJson(__DIR__ . '/../../foo.json', '$.test.value')]
        private int $val,
    ) {}
}
